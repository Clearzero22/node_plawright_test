⏺ 现在我明白了！这个步骤指南是绝对定位 (absolute bottom-6)
  在画布底部，它会挡住节点。问题是：

  1. 步骤指南在画布底部中央，使用 absolute 定位
  2. 节点的滚动容器 (overflow-auto) 可能让节点滚动到这个区域下方被遮挡

  有几个解决方案：

  方案1：移除步骤指南
  - 直接从 Canvas.tsx 中删除 <StepGuide />

  方案2：给画布底部增加内边距
  - 让节点不会被步骤指南遮挡

  方案3：让步骤指南可折叠/隐藏
  - 添加一个开关让用户可以隐藏它

  // src/hooks/usePlanWebSocket.ts
  function usePlanWebSocket(wsUrl: string) {
    const [connected, setConnected] = useState(false);
    const { updatePlan } = usePlanContext();

    useEffect(() => {
      const ws = new WebSocket(wsUrl);

      ws.onopen = () => setConnected(true);
      ws.onclose = () => setConnected(false);

      ws.onmessage = (event) => {
        const message: PlanUpdateMessage = JSON.parse(event.data);

        switch (message.type) {
          case 'plan_updated':
          case 'usage_updated':
            updatePlan(message.data);
            break;
          case 'subscription_expired':
            // 显示过期通知
            // 禁用相关功能
            break;
        }
      };

      // 自动重连
      ws.onclose = () => {
        setTimeout(() => reconnect(), 3000);
      };

      return () => ws.close();
    }, [wsUrl]);
  }

  数据获取与缓存

  // src/services/planService.ts
  class PlanService {
    private cache: PlanResponse | null = null;
    private cacheExpiry: number = 0;

    async getPlan(forceRefresh = false): Promise<PlanResponse> {
      // 检查缓存（5分钟有效期）
      if (!forceRefresh && this.cache && Date.now() < this.cacheExpiry) {
        return this.cache;
      }

      const response = await fetch('/api/plan');
      const data = await response.json();

      this.cache = data;

  // 统计数据
  export async function getDashboardStats() {
    const client = await pool.connect();

    try {
      // 总执行次数
      const totalResult = await client.query(
        'SELECT COUNT(*) as count FROM crawler_runs'
      );
      const totalExecutions = parseInt(totalResult.rows[0].count);

      // 成功率
      const successResult = await client.query(
        `SELECT 
          COUNT(*) FILTER (WHERE status = 'success') * 100.0 / COUNT(*) as rate
         FROM crawler_runs`
      );
      const successRate = parseFloat(successResult.rows[0].rate);

      // 发布商品数
      const productsResult = await client.query(
        'SELECT COUNT(DISTINCT asin) as count FROM clean_products'
      );
      const publishedProducts = parseInt(productsResult.rows[0].count);

      // 运行时长（秒）
      const durationResult = await client.query(
        'SELECT SUM(EXTRACT(EPOCH FROM (ended_at - started_at))) as total FROM 
  crawler_runs'
      );
      const totalDuration = Math.round((parseInt(durationResult.rows[0].total)
  || 0) / 3600);

         FROM crawler_runs

  // 最近执行记录
  export async function getRecentExecutions(limit: number) {
    const client = await pool.connect();

    try {
      const result = await client.query(
        `SELECT 
          id,
          workflow_name as "workflowName",
          status,
          EXTRACT(EPOCH FROM (ended_at - started_at)) as "duration",
          error_message as "errorMessage",
          started_at as "startedAt"
         FROM crawler_runs
         ORDER BY started_at DESC
         LIMIT $1`,
        [limit]
      );

      return result.rows.map(row => ({
        ...row,
        duration: Math.round(row.duration),
        startedAt: new Date(row.startedAt).toISOString(),
      }));
    } finally {
      client.release();
    }
  }

  // 本周趋势
  export async function getWeeklyTrend() {
    const client = await pool.connect();

    try {
      const result = await client.query(
        `SELECT 
          DATE(started_at) as date,
          COUNT(*) as count
         FROM crawler_runs
         WHERE started_at >= NOW() - INTERVAL '7 days'
         GROUP BY DATE(started_at)
         ORDER BY date`
      );

      return result.rows.map(row => ({
        date: row.date.toISOString().split('T')[0],
        count: row.count,
      }));
    } finally {
      client.release();
    }
  }
