## 📝 测试脚本模板

### 基础模板

```typescript
import { launchPersistent, getPageFromContext, log, sleep, checkLoginStatus, screenshot } from './utils';

async function test() {
  log('🚀 开始测试...', 'info');

  try {
    // 1. 启动浏览器
    const context = await launchPersistent();
    log('✅ 浏览器已启动', 'success');

    // 2. 获取页面
    const page = await getPageFromContext(context);

    // 3. 访问网站
    const url = 'https://example.com';
    log(`🌐 正在打开 ${url}...`, 'info');
    await page.goto(url, {
      timeout: 30000,
      waitUntil: 'domcontentloaded'
    });

    // 4. 获取页面信息
    const title = await page.title();
    log(`📄 页面标题: "${title}"`, 'info');

    // 5. 等待页面加载
    await sleep(2000);

    // 6. 检查登录状态
    const isLoggedIn = await checkLoginStatus(page, url);
    if (isLoggedIn) {
      log('✅ 登录状态: 已登录', 'success');
    } else {
      log('⚠️  登录状态: 未登录', 'warning');
    }

    // 7. 截图
    await screenshot(page, 'example');
    log('📸 截图已保存', 'info');

    // 8. 成功
    log('✅ 测试完成！', 'success');
    log('💡 浏览器保持打开状态', 'info');

    // 保持浏览器打开
    await new Promise(() => {});

  } catch (error: any) {
    log('❌ 测试失败: ' + error.message, 'error');
    process.exit(1);
  }
}

test();
```

### 数据提取模板

```typescript
import { launchPersistent, getPageFromContext, log, screenshot } from './utils';

async function test() {
  log('🚀 开始数据提取...', 'info');

  try {
    const context = await launchPersistent();
    const page = await getPageFromContext(context);

    await page.goto('https://example.com');

    // 提取单个元素文本
    const title = await page.textContent('.title');
    log(`标题: ${title}`, 'info');

    // 提取多个元素
    const items = await page.$$eval('.item', elements =>
      elements.map(el => el.textContent)
    );
    log(`找到 ${items.length} 个项目`, 'info');

    // 提取属性
    const href = await page.getAttribute('.link', 'href');
    log(`链接: ${href}`, 'info');

    // 提取复杂结构
    const data = await page.evaluate(() => {
      return {
        title: document.querySelector('.title')?.textContent,
        url: window.location.href,
        timestamp: Date.now()
      };
    });

    log('📊 数据:', 'info');
    console.log(data);

    // 保存到文件
    const fs = require('fs');
    fs.writeFileSync('output/data.json', JSON.stringify(data, null, 2));
    log('📁 数据已保存', 'success');

    await screenshot(page, 'data');
    await new Promise(() => {});

  } catch (error: any) {
    log('❌ 提取失败: ' + error.message, 'error');
    process.exit(1);
  }
}

test();
```

## 🛠️ 工具函数模板

### 登录检测函数

```typescript
export async function checkLoginStatus(page: Page, url: string): Promise<boolean> {
  try {
    if (url.includes('example.com')) {
      // 已登录标志
      const userMenu = await page.$('.user-info, .user-dropdown');
      // 未登录标志
      const loginButton = await page.$('a[href*="login"], .btn-login');

      return !!userMenu && !loginButton;
    }
    return false;
  } catch {
    return false;
  }
}
```

### 等待元素函数

```typescript
export async function waitForElement(page: Page, selector: string, timeout: number = 10000): Promise<boolean> {
  try {
    await page.waitForSelector(selector, { timeout });
    return true;
  } catch {
    return false;
  }
}
```

### 安全点击函数

```typescript
export async function safeClick(page: Page, selector: string): Promise<boolean> {
  try {
    await page.waitForSelector(selector, { timeout: 5000 });
    await page.click(selector);
    return true;
  } catch (error) {
    console.error(`点击失败: ${selector}`, error);
    return false;
  }
}
```

### 表单填充函数

export async function fillForm(page: Page, data: Record<string, string>): Promise<void> {
  for (const [selector, value] of Object.entries(data)) {
    await page.waitForSelector(selector, { timeout: 5000 });
    await page.fill(selector, value);
    console.log(`✅ 已填充: ${selector}`);
  }
}

## 📦 批量测试模板

```typescript
import { launchPersistent, getPageFromContext, log, sleep, checkLoginStatus, screenshot } from './utils';

const sites = [
  { name: 'example1', url: 'https://example1.com' },
  { name: 'example2', url: 'https://example2.com' },
  { name: 'example3', url: 'https://example3.com' },
];

async function testSite(site: typeof sites[0]) {
  log(`🚀 测试 ${site.name}...`, 'info');

  const context = await launchPersistent();
  const page = await getPageFromContext(context);

  try {
    await page.goto(site.url, { timeout: 30000 });
    const title = await page.title();
    log(`✅ ${site.name}: ${title}`, 'success');

    await screenshot(page, site.name);
    return true;
  } catch (error: any) {
    log(`❌ ${site.name}: ${error.message}`, 'error');
    return false;
  } finally {
    await page.close();
    await context.close();
  }
}

async function testAll() {
  const results = [];

  for (const site of sites) {
    const result = await testSite(site);
    results.push({ site, result });
    await sleep(1000); // 间隔1秒
  }

  // 打印总结
  const success = results.filter(r => r.result).length;
  log(`\n📊 结果: ${success}/${sites.length} 成功`, success === sites.length ? 'success' : 'warning');
}

testAll();
```

## 🎯 功能模块模板

### 搜索功能模块

```typescript
export class SearchModule {
  constructor(private page: Page) {}

  async search(keyword: string) {
    await this.page.fill('.search-input', keyword);
    await this.page.click('.search-button');
    await this.page.waitForSelector('.search-results');
  }

  async getResults() {
    return await this.page.$$eval('.result', items =>
      items.map(item => ({
        title: item.querySelector('.title')?.textContent,
        link: item.querySelector('a')?.href
      }))
    );
  }
}

// 使用
const search = new SearchModule(page);
await search.search('keyword');
const results = await search.getResults();
```

### 登录功能模块

```typescript
export class LoginModule {
  constructor(private page: Page) {}

  async login(username: string, password: string) {
    await this.page.goto('https://example.com/login');
    await this.page.fill('#username', username);
    await this.page.fill('#password', password);
    await this.page.click('.login-button');
    await this.page.waitForNavigation();
  }

  async logout() {
    await this.page.click('.logout-button');
    await this.page.waitForSelector('.login-button');
  }

  async isLoggedIn() {
    const userMenu = await this.page.$('.user-menu');
    return !!userMenu;
  }
}

// 使用
const login = new LoginModule(page);
await login.login('user', 'pass');
const loggedIn = await login.isLoggedIn();
```

## 🔄 错误处理模板

### 重试机制

```typescript
export async function retry<T>(
  fn: () => Promise<T>,
  maxRetries: number = 3,
  delay: number = 1000
): Promise<T> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      console.log(`重试 ${i + 1}/${maxRetries}...`);
      await sleep(delay);
    }
  }
  throw new Error('重试失败');
}

// 使用
const result = await retry(() => page.click('.button'));
```

### 超时控制

```typescript
export async function withTimeout<T>(
  fn: () => Promise<T>,
  timeout: number
): Promise<T> {
  return Promise.race([
    fn(),
    new Promise<T>((_, reject) =>
      setTimeout(() => reject(new Error('超时')), timeout)
    )
  ]);
}

// 使用
const result = await withTimeout(
  () => page.waitForSelector('.element'),
  5000
);
```
